#include <iostream>
#include <string>  
#include <algorithm> 
using namespace std;

bool sonAnagramas(string palabra1, string palabra2) {
    
    if (palabra1 == palabra2) {
        return false;
    }

    if (palabra1.length() != palabra2.length()) {
        return false;
    }


    std::sort(palabra1.begin(), palabra1.end());
    std::sort(palabra2.begin(), palabra2.end());


    return palabra1 == palabra2;
}
